class Solution {
    public List<List<Integer>> threeSum(int[] nums) {
        HashSet<List<Integer>> set = new HashSet<>();
        Arrays.sort(nums);
        for (int p1 =0;p1<nums.length;p1++) {
            int p2 = p1+1;
            int p3 = nums.length-1;
            int required = 0 - (nums[p1]);
            while (p2 < p3) {
            if (nums[p2] + nums[p3] == required) {
                List<Integer> list = new ArrayList<>();
                list.add(nums[p1]);
                list.add(nums[p2]);
                list.add(nums[p3]);
                set.add(list);
                p2++;
                p3--;
            }
            else if (nums[p2] + nums[p3] > required) {
                p3--;
            }
            else {
                p2++;
            }
            }
        }
        return new ArrayList<>(set);
    }
}
