```java
class Solution {
    public int[] twoSum(int[] nums, int target) {
        HashMap<Integer,Integer> map = new HashMap<>();
        for (int i =0;i<nums.length;i++) {
            map.put(nums[i],i);
        }

        for (int j = 0; j < nums.length; j++) {
            int currentIndex = j;
            int required = target - nums[currentIndex];
            if (map.containsKey(required) && map.get(required) != currentIndex) {
                int resultIndex = map.get(required);
                int[] result = {currentIndex,resultIndex};
                return result;
            }
        }
        int[] result = new int[2];
        return result;
    }
}
```
