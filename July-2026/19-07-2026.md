```java
class Solution {
    public boolean searchMatrix(int[][] matrix, int target) {
        int ilow = 0;
        int ihigh = matrix.length - 1;
        int jlow = 0;
        int jhigh = matrix[0].length - 1;
        while (ilow <= ihigh) {
            int imid = (ilow + ihigh) / 2;
            int jmid = (jlow + jhigh) / 2;
            if (matrix[imid][jmid] == target) {
                return true;
            }
            if (target <= matrix[imid][jhigh] && target >= matrix[imid][jlow]) {
                if (target < matrix[imid][jmid]) {
                    jhigh = jmid - 1;
                } else if (target > matrix[imid][jmid]) {
                    jlow = jmid + 1;
                }
            } else {
                if (target < matrix[imid][jlow]) {
                    ihigh = imid - 1;
                } else if (target > matrix[imid][jhigh]) {
                    ilow = imid + 1;
                }
            }
        }
        return false;
    }
}
```

```java
class Solution {
    public int findMin(int[] nums) {
        int low = 0;
        int high = nums.length-1;
        while (low <= high) {
            int mid = (low+high)/2;
            if (nums[low] <= nums[high]) {
                return nums[low];
            }

            if (nums[low] <= nums[mid]) {
                low = mid+1;
            }
            else {
                high = mid;
            }
        }
        return low;
    }
}
```
