27. Remove Element
Easy

class Solution {
    public int removeElement(int[] nums, int val) {
        int write = 0;
        int read = 0;
        while (read < nums.length) {
            if (nums[read] != val) {
                nums[write] = nums[read];
                write++;
                read++;
            }
            else {
                read++;
            }
        }
        return write;
    }
}

905. Sort Array By Parity
Easy

class Solution {
    public int[] sortArrayByParity(int[] nums) {
        int read = 0;
        int write = 0;

        while (read < nums.length) {
            if (nums[read] % 2 ==0) {
                int temp = nums[write];
                nums[write] = nums[read];
                nums[read] = temp;
                write++;
            }
            read++;
        }
        return nums;
    }
}
