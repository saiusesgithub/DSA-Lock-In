```java
class Solution {
    public int mySqrt(int x) {
        long low = 0;
        long high = x;
        while (low <= high) {
            long mid = (low + high) / 2;
            long square = mid*mid;
            if (square == x) {
                return (int)mid;
            }
            else if (x < square) {
                high = mid-1;
            }
            else {
                low = mid+1;
            }
        }
        return (int)high;
    }
}
```

class Solution {
    public boolean isPerfectSquare(int num) {
        long low = 0;
        long high = num;
        while (low <= high) {
            long mid = (low+high)/2;
            long square = mid * mid;
            if (square == num) {
                return true;
            }
            else if (num < square) {
                high = mid-1;
            }
            else {
                low = mid+1;
            }
            
        }
        return false;
    }
}
