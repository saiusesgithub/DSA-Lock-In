class MyStack {
    Queue<Integer> q;
    public MyStack() {
        q = new LinkedList<>();
    }
    
    public void push(int x) {
        q.offer(x);
    }
    
    public int pop() {
        Queue<Integer> temp = new LinkedList<>();
        int queuesize = q.size();
        for (int i = 0;i < queuesize;i++) {
            if (queuesize-2 <0) {
                return q.poll();
            }
            if (i == queuesize-2) {
                temp.offer(q.poll());
                int result = q.poll();
                q = temp;
                return result;
            }
            temp.offer(q.poll());
        }
        return -1;
    }
    
    public int top() {
        Queue<Integer> temp = new LinkedList<>();
        int qsize = q.size();
        for (int i = 0;i < qsize;i++) {
            if (i == qsize-1) {
                int result = q.poll();
                q = temp;
                q.offer(result);
                return result;
            }
            temp.offer(q.poll()); 
        }
        return -1;
    }
    
    public boolean empty() {
        return q.isEmpty();
    }
}
